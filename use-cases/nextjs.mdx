Add a PDF endpoint to a Next.js App Router project.

## Setup

```bash
npm install rendio
```

```bash
# .env.local
RENDIO_API_KEY=rk_live_xxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxx
```

## API route

```ts
// app/api/pdf/route.ts
import { NextRequest, NextResponse } from "next/server";
import Rendio, {
  RendioError,
  AuthenticationError,
  ValidationError,
  QuotaExceededError,
  PlanLimitError,
} from "rendio";

const rendio = new Rendio(); // reads RENDIO_API_KEY — module-scope so it's reused

export async function POST(req: NextRequest) {
  try {
    const { html, data, pageSize } = (await req.json()) as {
      html: string;
      data?: Record<string, unknown>;
      pageSize?: "A4" | "Letter" | "A3" | "Legal" | "Tabloid";
    };

    const pdf = await rendio.pdfs.create({ html, data, pageSize });
    return NextResponse.json(pdf);
  } catch (err) {
    if (err instanceof AuthenticationError) {
      console.error("[rendio] API key invalid or revoked");
      return NextResponse.json(
        { error: "PDF service unavailable" },
        { status: 500 },
      );
    }
    if (err instanceof ValidationError) {
      return NextResponse.json(
        { error: err.message, code: err.code },
        { status: 400 },
      );
    }
    if (err instanceof QuotaExceededError) {
      return NextResponse.json(
        { error: "Rendio quota exhausted", code: err.code },
        { status: 503 },
      );
    }
    if (err instanceof PlanLimitError) {
      return NextResponse.json(
        { error: `Feature "${err.feature}" requires a higher plan`, code: err.code },
        { status: 402 },
      );
    }
    if (err instanceof RendioError) {
      console.error("[rendio]", {
        code: err.code,
        requestId: err.requestId,
        traceId: err.traceId,
      });
      return NextResponse.json(
        { error: err.message, code: err.code },
        { status: err.statusCode ?? 500 },
      );
    }
    throw err;
  }
}
```

## Client usage

```ts
// lib/generate-pdf.ts
export async function generatePdf(html: string, data?: Record<string, unknown>) {
  const res = await fetch("/api/pdf", {
    method: "POST",
    headers: { "Content-Type": "application/json" },
    body: JSON.stringify({ html, data, pageSize: "A4" }),
  });

  if (!res.ok) {
    const { error } = await res.json();
    throw new Error(error);
  }

  return (await res.json()) as { id: string; url: string; pages: number; size: number };
}
```

<Tip>
  The `Rendio` client is instantiated at module scope, not inside the
  handler. Reusing it across requests avoids per-request setup and keeps
  keep-alive connections warm.
</Tip>

## Edge runtime

Rendio's SDK works on the Edge runtime (uses global `fetch`). Add:

```ts
export const runtime = "edge";
```

at the top of the route file.

<Warning>
OpenTelemetry tracing uses Node's `AsyncLocalStorage`, which is stubbed on
the Edge runtime. Trace propagation will be limited — if you rely on APM
correlation, stick to the Node runtime.
</Warning>

## Streaming buffer mode

When you want to stream raw PDF bytes straight to the browser (no CDN hop):

```ts
export async function POST(req: NextRequest) {
  try {
    const { html } = await req.json();
    const result = await rendio.pdfs.create({ html, response: "buffer" });

    return new Response(result.buffer, {
      status: 200,
      headers: {
        "Content-Type": "application/pdf",
        "Content-Disposition": 'attachment; filename="invoice.pdf"',
      },
    });
  } catch (err) {
    if (err instanceof RendioError) {
      return NextResponse.json(
        { error: err.message, code: err.code },
        { status: err.statusCode ?? 500 },
      );
    }
    throw err;
  }
}
```

## Server Action

Render a PDF and return the CDN URL to the client:

```ts
// app/actions/generate-pdf.ts
"use server";

import Rendio from "rendio";

const rendio = new Rendio();

export async function generateInvoicePdf(html: string) {
  const pdf = await rendio.pdfs.create({ html, pageSize: "A4" });
  return { url: pdf.url, id: pdf.id };
}
```

```tsx
// app/invoice/page.tsx
"use client";

import { generateInvoicePdf } from "../actions/generate-pdf";
import { useState } from "react";

export default function InvoicePage() {
  const [url, setUrl] = useState<string | null>(null);

  async function handleGenerate() {
    const { url } = await generateInvoicePdf("<h1>Invoice #001</h1>");
    setUrl(url);
  }

  return (
    <div>
      <button onClick={handleGenerate}>Generate PDF</button>
      {url && <a href={url} download>Download PDF</a>}
    </div>
  );
}
```

## Browser download (client-side)

When your API route returns buffer-mode bytes, trigger a file download in
the browser:

```ts
async function downloadPdf(html: string) {
  const res = await fetch("/api/pdf", {
    method: "POST",
    headers: { "Content-Type": "application/json" },
    body: JSON.stringify({ html }),
  });

  if (!res.ok) throw new Error("PDF generation failed");

  const blob = await res.blob();
  const url = URL.createObjectURL(blob);
  const a = document.createElement("a");
  a.href = url;
  a.download = "document.pdf";
  a.click();
  URL.revokeObjectURL(url);
}
```
